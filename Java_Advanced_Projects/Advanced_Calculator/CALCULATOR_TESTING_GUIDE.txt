#!/bin/bash

# CALCULATOR TESTING GUIDE
# ========================
#
# This file is both a readable guide and a runnable test shortcut.
# To run all tests, open a terminal in this folder and type:
#
#   bash CALCULATOR_TESTING_GUIDE.txt
#
# The commands below execute Calculator with test input values.
#
# If you prefer, you can still read the guide manually from this file.

print_section() {
  local title="$1"
  echo
  echo "============================================================"
  echo "${title}"
  echo "============================================================"
}

print_case() {
  local title="$1"
  echo
  echo "--- ${title} ---"
}

print_section "TEST 1: BASIC ARITHMETIC"
print_case "Testing Addition (5 + 10 = 15)"
echo -e "5\n10\n1\n1\nn\nn" | java Calculator

print_case "Testing Subtraction (10 - 5 = 5)"
echo -e "10\n5\n1\n2\nn\nn" | java Calculator

print_case "Testing Multiplication (5 * 10 = 50)"
echo -e "5\n10\n1\n3\nn\nn" | java Calculator

print_case "Testing Division (10 / 2 = 5)"
echo -e "10\n2\n1\n4\nn\nn" | java Calculator

print_case "Testing Power (2 ^ 3 = 8)"
echo -e "2\n3\n1\n5\nn\nn" | java Calculator

print_case "Testing Modulus (10 % 3 = 1)"
echo -e "10\n3\n1\n12\nn\nn" | java Calculator

print_case "Testing Square Root (√9 = 3)"
echo -e "9\n0\n1\n9\nn\nn" | java Calculator

print_case "Testing Percentage (%1: 50% of 200 = 100)"
echo -e "50\n200\n1\n6\nn\nn" | java Calculator

print_section "TEST 2: COMPARISON OPERATIONS"
print_case "Testing Equal (5 == 5 = true)"
echo -e "5\n5\n2\n1\nn\nn" | java Calculator

print_case "Testing Greater Than (10 > 5 = true)"
echo -e "10\n5\n2\n2\nn\nn" | java Calculator

print_case "Testing Less Than (5 < 10 = true)"
echo -e "5\n10\n2\n3\nn\nn" | java Calculator

print_case "Testing Not Equal (5 != 10 = true)"
echo -e "5\n10\n2\n6\nn\nn" | java Calculator

print_section "TEST 3: MEMORY OPERATIONS"
print_case "Testing Memory Store and Recall"
echo -e "5\n10\n1\n1\ny\n2\ny\n5\nn\nn" | java Calculator

print_case "Testing Memory Clear"
echo -e "5\n10\n1\n1\ny\n2\ny\n1\ny\n5\nn\nn" | java Calculator

print_case "Testing Memory Add (M+)"
echo -e "5\n5\n1\n1\ny\n2\ny\n3\nn\nn" | java Calculator

print_case "Testing Memory Subtract (M-)"
echo -e "7\n3\n1\n2\ny\n2\ny\n4\nn\nn" | java Calculator

print_case "Testing Memory Store Number (MSN)"

print_section "TEST 4: EDGE CASES"
print_case "Testing Division by Zero (should show Infinity)"
echo -e "10\n0\n1\n4\nn\nn" | java Calculator

print_case "Testing Negative Numbers (-5 + 10 = 5)"
echo -e "-5\n10\n1\n1\nn\nn" | java Calculator

print_case "Testing Decimal Numbers (3.5 * 2 = 7.0)"
echo -e "3.5\n2\n1\n3\nn\nn" | java Calculator

print_section "TEST 5: ERROR HANDLING"
print_case "Testing Invalid Menu Choice (choice 3, should show error)"
echo -e "5\n10\n3\nn\nn" | java Calculator
